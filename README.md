# Tuple-and-List-

t = ["mayur", 10, 1.5]
print(t)

# Indexing the tuple
t[0]
print(t[0])

t[-2]
print(t[-2])


# Concantenation the tuple
t2 = t + ["jagtap", 5.73]
print(t2)

# Slicing the tuple
t2[2:4]
print(t2[2:4])

# Length of tuple
print(len(t2))

# Sorting the tuple
Ratings = (0, 9, 6, 5, 10, 8, 9, 6, 2)
print(Ratings)
RatingsSorted = sorted(Ratings)
print(RatingsSorted)

# List


l = ["mayur jagtap", 10.4, 2005]
print(l)

# Indexing the list
print(l[1])

# Slicing
print(l[1:3])

# Concantenation the list
l1 = l + ["harry", 45]
print(l1)

# Append and extend
l.append(["pop", 34])
